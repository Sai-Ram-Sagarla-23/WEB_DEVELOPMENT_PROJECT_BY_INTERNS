// Modern async/await pattern async function fetchUser(id) { try { const resp = await fetch(`/api/users/${id}`, { method: 'GET', headers: { 'Authorization': `Bearer ${token}` } }); if (!resp.ok) throw new Error(`HTTP ${resp.status}`); return await resp.json(); } catch (err) { console.error('Fetch failed:', err); throw err; } } // Parallel requests (faster!) const [users, posts, comments] = await Promise.all([ fetch('/api/users').then(r => r.json()), fetch('/api/posts').then(r => r.json()), fetch('/api/comments').then(r => r.json()), ]); // AbortController (cancel requests) const ctrl = new AbortController(); fetch(url, { signal: ctrl.signal }); setTimeout(() => ctrl.abort(), 5000); // 5s timeout // Retry with exponential backoff async function fetchWithRetry(url, tries=3) { for (let i=0; i<tries; i++) { try { return await fetch(url).then(r=>r.json()); } catch(e) { if (i===tries-1) throw e; await new Promise(res=>setTimeout(res, 2**i*1000)); } } }

// React 19 — Modern patterns import { useState, useEffect, useCallback, useMemo } from 'react'; // Custom Hook — reusable logic function useFetch(url) { const [data, setData] = useState(null); const [loading, setLoading] = useState(true); const [error, setError] = useState(null); useEffect(() => { let cancelled = false; fetch(url) .then(r => r.json()) .then(d => !cancelled && setData(d)) .catch(e => !cancelled && setError(e)) .finally(() => !cancelled && setLoading(false)); return () => { cancelled = true; }; // cleanup }, [url]); return { data, loading, error }; } // Component function UserList() { const { data, loading, error } = useFetch('/api/users'); const [filter, setFilter] = useState(''); const filtered = useMemo(() => data?.filter(u => u.name.includes(filter)), [data, filter]); // only recomputes when deps change if (loading) return <Spinner />; if (error) return <Error message={error.message} />; return ( <div> <input value={filter} onChange={e => setFilter(e.target.value)} /> {filtered.map(u => <UserCard key={u.id} user={u} />)} </div> ); }
